## JavaScript / Node.js

### Ürün Listesi Çekme

```javascript
const axios = require('axios');

const API_BASE = 'https://api.example.com/v1';
const API_KEY = 'your-api-key';

async function getProducts(page = 1, limit = 20) {
  const response = await axios.get(`${API_BASE}/products`, {
    headers: {
      'X-API-Key': API_KEY
    },
    params: { page, limit }
  });

  return response.data;
}

// Kullanım
const products = await getProducts(1, 50);
console.log(products.data);
```

### Sipariş Oluşturma

```javascript
async function createOrder(orderData) {
  const response = await axios.post(`${API_BASE}/orders`, orderData, {
    headers: {
      'X-API-Key': API_KEY,
      'Content-Type': 'application/json'
    }
  });

  return response.data;
}

// Kullanım
const order = await createOrder({
  customer: {
    email: 'customer@example.com',
    phone: '5551234567'
  },
  items: [
    { productId: 'PROD-001', quantity: 2 },
    { productId: 'PROD-002', quantity: 1 }
  ],
  shippingAddress: {
    city: 'İstanbul',
    district: 'Kadıköy',
    address: 'Moda Cad. No:1'
  }
});
```

## PHP

### Ürün Güncelleme

```php
<?php

class EcommerceAPI {
    private $apiKey;
    private $baseUrl = 'https://api.example.com/v1';

    public function __construct($apiKey) {
        $this->apiKey = $apiKey;
    }

    public function updateProduct($productId, $data) {
        $ch = curl_init();

        curl_setopt_array($ch, [
            CURLOPT_URL => "{$this->baseUrl}/products/{$productId}",
            CURLOPT_CUSTOMREQUEST => 'PUT',
            CURLOPT_POSTFIELDS => json_encode($data),
            CURLOPT_HTTPHEADER => [
                'X-API-Key: ' . $this->apiKey,
                'Content-Type: application/json'
            ],
            CURLOPT_RETURNTRANSFER => true
        ]);

        $response = curl_exec($ch);
        curl_close($ch);

        return json_decode($response, true);
    }
}

// Kullanım
$api = new EcommerceAPI('your-api-key');
$result = $api->updateProduct('PROD-001', [
    'price' => 299.99,
    'stock' => 150
]);
```

## Python

### Stok Senkronizasyonu

```python
import requests

class EcommerceClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = 'https://api.example.com/v1'

    def _headers(self):
        return {
            'X-API-Key': self.api_key,
            'Content-Type': 'application/json'
        }

    def update_stock(self, sku, quantity):
        response = requests.patch(
            f'{self.base_url}/products/{sku}/stock',
            json={'quantity': quantity},
            headers=self._headers()
        )
        return response.json()

    def bulk_update_stock(self, updates):
        """
        updates: [{'sku': 'ABC', 'quantity': 10}, ...]
        """
        response = requests.post(
            f'{self.base_url}/products/stock/bulk',
            json={'items': updates},
            headers=self._headers()
        )
        return response.json()

# Kullanım
client = EcommerceClient('your-api-key')

# Tekli güncelleme
client.update_stock('SKU-001', 50)

# Toplu güncelleme
client.bulk_update_stock([
    {'sku': 'SKU-001', 'quantity': 50},
    {'sku': 'SKU-002', 'quantity': 30},
    {'sku': 'SKU-003', 'quantity': 0}
])
```

## cURL

### Temel İstekler

```bash
# Ürün listesi
curl -X GET "https://api.example.com/v1/products?page=1&limit=20" \
  -H "X-API-Key: your-api-key"

# Ürün detayı
curl -X GET "https://api.example.com/v1/products/PROD-001" \
  -H "X-API-Key: your-api-key"

# Sipariş oluştur
curl -X POST "https://api.example.com/v1/orders" \
  -H "X-API-Key: your-api-key" \
  -H "Content-Type: application/json" \
  -d '{
    "customer": {"email": "test@example.com"},
    "items": [{"productId": "PROD-001", "quantity": 1}]
  }'
```

## Webhook Alma

### Node.js Express

```javascript
const express = require('express');
const crypto = require('crypto');

const app = express();
app.use(express.json());

function verifyWebhook(payload, signature, secret) {
  const computed = crypto
    .createHmac('sha256', secret)
    .update(JSON.stringify(payload))
    .digest('hex');
  return computed === signature;
}

app.post('/webhook', (req, res) => {
  const signature = req.headers['x-webhook-signature'];

  if (!verifyWebhook(req.body, signature, 'webhook-secret')) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const { event, data } = req.body;

  switch (event) {
    case 'order.created':
      console.log('Yeni sipariş:', data.orderId);
      break;
    case 'order.paid':
      console.log('Ödeme alındı:', data.orderId);
      break;
  }

  res.status(200).json({ received: true });
});
```
